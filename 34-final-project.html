Проект 1: Игра «Угадай число»

Компьютер загадывает число от 1 до 100, а игрок пытается его угадать. После каждой попытки компьютер даёт подсказку: «больше» или «меньше».

Используемые темы: переменные, ввод/вывод, random, условный оператор, цикл while, обработка ошибок.

import random

def guess_the_number():
    """Игра «Угадай число»."""
    secret = random.randint(1, 100)
    attempts = 0
    max_attempts = 10

    print("Я загадал число от 1 до 100. У тебя 10 попыток!")

    while attempts < max_attempts:
        try:
            guess = int(input(f"Попытка {attempts + 1}/{max_attempts}: "))
        except ValueError:
            print("Пожалуйста, введи целое число.")
            continue

        attempts += 1

        if guess < secret:
            print("Больше!")
        elif guess > secret:
            print("Меньше!")
        else:
            print(f"Поздравляю! Ты угадал число {secret} за {attempts} попыток!")
            return

    print(f"К сожалению, попытки закончились. Я загадал число {secret}.")

# Запуск игры
guess_the_number()

Проект 2: Многофункциональный калькулятор

Калькулятор с поддержкой базовых операций, историей вычислений и обработкой ошибок.

Используемые темы: функции, словари, циклы, строки, обработка ошибок, форматирование.

def calculator():
    """Интерактивный калькулятор с историей."""
    history = []  # Список для хранения истории

    while True:
        print("\n" + "=" * 30)
        print("КАЛЬКУЛЯТОР")
        print("=" * 30)
        print("1. Сложение (+)")
        print("2. Вычитание (-)")
        print("3. Умножение (*)")
        print("4. Деление (/)")
        print("5. Возведение в степень (**)")
        print("6. Показать историю")
        print("7. Очистить историю")
        print("0. Выход")

        choice = input("\nВыбери операцию: ")

        if choice == "0":
            print("До свидания!")
            break
        elif choice == "6":
            if history:
                print("\nИстория вычислений:")
                for i, record in enumerate(history, 1):
                    print(f"  {i}. {record}")
            else:
                print("\nИстория пуста.")
            continue
        elif choice == "7":
            history.clear()
            print("История очищена.")
            continue

        if choice not in ["1", "2", "3", "4", "5"]:
            print("Неверный выбор. Попробуй снова.")
            continue

        try:
            a = float(input("Введи первое число: "))
            b = float(input("Введи второе число: "))
        except ValueError:
            print("Ошибка: нужно ввести число.")
            continue

        if choice == "1":
            result = a + b
            op = "+"
        elif choice == "2":
            result = a - b
            op = "-"
        elif choice == "3":
            result = a * b
            op = "*"
        elif choice == "4":
            if b == 0:
                print("Ошибка: деление на ноль!")
                continue
            result = a / b
            op = "/"
        elif choice == "5":
            result = a ** b
            op = "**"

        record = f"{a} {op} {b} = {result:.2f}"
        print(f"Результат: {record}")
        history.append(record)

# Запуск калькулятора
calculator()

Проект 3: Анализатор текста

Программа, которая анализирует введённый текст: считает количество слов, предложений, символов и самые частые слова.

Используемые темы: строки, методы строк, списки, словари, list comprehensions, сортировка, срезы.

def text_analyzer(text: str) -> dict:
    """Анализирует текст и возвращает статистику."""
    # Количество символов (с пробелами)
    chars_total = len(text)

    # Количество символов без пробелов
    chars_no_spaces = len(text.replace(" ", ""))

    # Количество слов
    words = text.split()
    word_count = len(words)

    # Количество предложений (по знакам .!?)
    sentences = [s for s in text.replace("!", ".").replace("?", ".").split(".") if s.strip()]
    sentence_count = len(sentences)

    # Частота слов
    word_freq = {}
    for word in words:
        word = word.lower().strip(".,!?:;\"'()")
        if word:
            word_freq[word] = word_freq.get(word, 0) + 1

    # Топ-5 самых частых слов
    top_words = sorted(word_freq.items(), key=lambda x: x[1], reverse=True)[:5]

    return {
        "symbols_total": chars_total,
        "symbols_no_spaces": chars_no_spaces,
        "words": word_count,
        "sentences": sentence_count,
        "top_words": top_words
    }

# Пример использования
sample_text = """Python — отличный язык для начинающих.
Он простой, но мощный. Python используется в веб-разработке,
науке о данных и искусственном интеллекте.
Python, Python, Python!"""

stats = text_analyzer(sample_text)

print("=" * 40)
print("АНАЛИЗ ТЕКСТА")
print("=" * 40)
print(f"Символов (всего):      {stats['symbols_total']}")
print(f"Символов (без пробелов): {stats['symbols_no_spaces']}")
print(f"Слов:                  {stats['words']}")
print(f"Предложений:           {stats['sentences']}")
print("\nТоп-5 самых частых слов:")
for word, count in stats["top_words"]:
    print(f"  '{word}' — {count} раз(а)")

Проект 4: ToDo-приложение (на классах)

Этот проект закрепляет ООП и работу с файлами. Создадим консольное приложение для управления списком задач с сохранением в файл.

Техническое задание

  1. Класс Task — хранит описание задачи, статус (сделано/не сделано) и дату создания
  2. Класс TodoList — хранит список задач, умеет добавлять, удалять, отмечать выполненными, показывать список
  3. Сохранение задач в текстовый файл tasks.txt и загрузка при старте
  4. Консольное меню с командами: добавить, показать, отметить, удалить, выйти

Код проекта

from datetime import datetime
import os

class Task:
    """Одна задача."""
    def __init__(self, description, done=False, created=None):
        self.description = description
        self.done = done
        self.created = created if created else datetime.now()
    
    def mark_done(self):
        self.done = True
    
    def __str__(self):
        status = "✓" if self.done else "✗"
        return f"[{status}] {self.description} ({self.created.strftime('%d.%m.%Y %H:%M')})"
    
    def to_line(self):
        """Сериализация для сохранения в файл."""
        return f"{self.description}|{self.done}|{self.created.isoformat()}\n"


class TodoList:
    """Список задач."""
    def __init__(self, filename="tasks.txt"):
        self.filename = filename
        self.tasks = []
        self.load()
    
    def add(self, description):
        task = Task(description)
        self.tasks.append(task)
        print(f"Добавлена задача #{len(self.tasks)}: {description}")
    
    def show(self):
        if not self.tasks:
            print("Список задач пуст.")
            return
        for i, task in enumerate(self.tasks, 1):
            print(f"{i}. {task}")
    
    def mark(self, index):
        if 1 <= index <= len(self.tasks):
            self.tasks[index - 1].mark_done()
            print(f"Задача #{index} отмечена как выполненная!")
        else:
            print("Неверный номер задачи.")
    
    def delete(self, index):
        if 1 <= index <= len(self.tasks):
            removed = self.tasks.pop(index - 1)
            print(f"Задача «{removed.description}» удалена.")
        else:
            print("Неверный номер задачи.")
    
    def save(self):
        with open(self.filename, "w", encoding="utf-8") as f:
            for task in self.tasks:
                f.write(task.to_line())
        print(f"Сохранено {len(self.tasks)} задач в {self.filename}")
    
    def load(self):
        if not os.path.exists(self.filename):
            return
        with open(self.filename, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                parts = line.split("|")
                if len(parts) == 3:
                    desc, done_str, created_str = parts
                    task = Task(
                        description=desc,
                        done=(done_str == "True"),
                        created=datetime.fromisoformat(created_str)
                    )
                    self.tasks.append(task)
        if self.tasks:
            print(f"Загружено {len(self.tasks)} задач из {self.filename}")


def main():
    todo = TodoList()
    while True:
        print("\n" + "=" * 40)
        print("📋 ToDo-приложение")
        print("1. Показать задачи")
        print("2. Добавить задачу")
        print("3. Отметить как выполненную")
        print("4. Удалить задачу")
        print("5. Сохранить и выйти")
        print("=" * 40)
        
        choice = input("Выберите действие (1-5): ").strip()
        
        if choice == "1":
            todo.show()
        elif choice == "2":
            desc = input("Введите описание задачи: ").strip()
            if desc:
                todo.add(desc)
        elif choice == "3":
            todo.show()
            try:
                idx = int(input("Номер задачи для отметки: "))
                todo.mark(idx)
            except ValueError:
                print("Введите число!")
        elif choice == "4":
            todo.show()
            try:
                idx = int(input("Номер задачи для удаления: "))
                todo.delete(idx)
            except ValueError:
                print("Введите число!")
        elif choice == "5":
            todo.save()
            print("До свидания!")
            break
        else:
            print("Неверный выбор. Попробуйте 1-5.")

if __name__ == "__main__":
    main()
